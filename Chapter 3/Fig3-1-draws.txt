

lambda <- seq(0.1, 3, 0.1) 

n<-length(lambda)
T<-100000

pdraw<-numeric(n)

for(i in 1:n){
	X<-rpois(T,lambda[i])
	Y<-rpois(T,lambda[i])
	pdraw[i]<- mean(X==Y) 
} 

plot(lambda, pdraw, type='l', bty="l", xlab="Expected goals", ylab="Probability of a draw", lwd=2, col="blue")

pdf("fig1-draws.pdf", width=11, height=8)
plot(lambda, pdraw, type='l', bty="l", xlab="Expected goals", ylab="Probability of a draw", lwd=2, col="blue")
graphics.off()
